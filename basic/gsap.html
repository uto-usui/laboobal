# motion path

import { gsap, MotionPathPlugin } from 'gsap/all'
gsap.registerPlugin(MotionPathPlugin)

const { target, path } = this.$refs

gsap.set(target, {
  xPercent: -50,
  yPercent: -50,
  transformOrigin: '50% 50%',
})

gsap.to(target, {
  duration: 10,
  repeat: -1,
  yoyo: true,
  ease: 'power2.inOut',
  motionPath:{
    autoRotate: true,
    immediateRender: true,
    path: path,
    align: path,
  }
})
